use core::num::Wrapping;

pub fn calculate_crc16(data: &[u8]) -> u16 {
    let mut crc: u16 = 0xffff;

    for &byte in data {
        let ix = (crc >> 8) ^ u16::from(byte);
        crc = (crc << 8) ^ CRC16_TABLE[ix as usize];
    }

    crc
}

/// A simple checksum for validating a packet's header.
pub fn calculate_header_lrc(header: &[u8]) -> u8 {
    debug_assert!(header.len() >= 4);

    // All these Wrappings kinda makes me sad...
    let header_sum = Wrapping(header[0])
        + Wrapping(header[1])
        + Wrapping(header[2])
        + Wrapping(header[3]);
    let lrc = (header_sum ^ Wrapping(0xFF)) + Wrapping(1);

    lrc.0
}
