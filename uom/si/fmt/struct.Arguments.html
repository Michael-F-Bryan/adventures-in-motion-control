A struct to specify a display style and unit.

Usage

Indirect style

let l = Length::new::<meter>(1.0);
let a = Length::format_args(centimeter, Description);

Direct style

let l = Length::new::<meter>(1.0);
let a = l.into_format_args(centimeter, Description);
